117 bool cISLogger::InitSaveCommon(eLogType logType, const string& directory, const string& subDirectory, int numDevices, float maxDiskSpacePercent, uint32_t maxFileSize, uint32_t maxChunkSize, bool useSubFolderTimestamp)
118 {
119  static const int minFileCount = 50;
120  static const int maxFileCount = 10000;
121 
122  // Close any open files
123  CloseAllFiles();
124 
125  m_logType = logType;
126  m_directory = (directory.empty() ? DEFAULT_LOGS_DIRECTORY : directory);
127  maxDiskSpacePercent = _CLAMP(maxDiskSpacePercent, 0.01f, 0.99f);
128  uint64_t availableSpace = cISLogger::GetDirectorySpaceAvailable(m_directory);
129  m_maxDiskSpace = (maxDiskSpacePercent <= 0.0f ? availableSpace : (uint64_t)(availableSpace * maxDiskSpacePercent));
130 
131  // ensure there are between min and max file count
132  if (maxFileSize > m_maxDiskSpace / minFileCount)
133  {
134  m_maxFileSize = (uint32_t)(m_maxDiskSpace / minFileCount);
135  }
136  else if (maxFileSize < m_maxDiskSpace / maxFileCount)
137  {
138  m_maxFileSize = (uint32_t)(m_maxDiskSpace / maxFileCount);
139  }
140  else
141  {
142  m_maxFileSize = maxFileSize;
143  }
144 
145  // make sure chunk size is valid
146  if (maxChunkSize > m_maxFileSize)
147  {
148  m_maxChunkSize = _MIN(m_maxFileSize, MAX_CHUNK_SIZE);
149  }
150  else
151  {
152  m_maxChunkSize = _MIN(maxChunkSize, MAX_CHUNK_SIZE);
153  }
154 
155  // create root dir
156  _MKDIR(m_directory.c_str());
157 
158  if (useSubFolderTimestamp)
159  {
160  // create time stamp dir
161  m_directory = directory + "/" + m_timeStamp;
162  _MKDIR(m_directory.c_str());
163 
164  if (!subDirectory.empty())
165  {
166  // create sub dir
167  m_directory += "/" + subDirectory;
168  _MKDIR(m_directory.c_str());
169  }
170  }
171 
172  // create empty stats file to track timestamps
173  fstream fs;
174  fs.open((m_directory + "/" + subDirectory + "/stats.txt").c_str(), ios::out);
175  fs.close();
176 
177  // Initialize devices
178  return InitDevicesForWriting(numDevices);
179 }

260 bool cISLogger::GetAllFilesInDirectory(const string& directory, bool recursive, const string& regexPattern, vector<string>& files)
261 {
262  size_t startSize = files.size();
263  regex* rePtr = NULL;
264  regex re;
265  if (regexPattern.length() != 0)
266  {
267  re = regex(regexPattern, regex::icase);
268  rePtr = &re;
269  }
270 
271 #if PLATFORM_IS_WINDOWS
272 
273  HANDLE dir;
274  WIN32_FIND_DATAA file_data;
275  if ((dir = FindFirstFileA((directory + "/*").c_str(), &file_data)) == INVALID_HANDLE_VALUE)
276  {
277  return false;
278  }
279  do
280  {
281  string file_name = file_data.cFileName;
282  string full_file_name = directory + "/" + file_name;
283  if (file_name[0] == '.' || (rePtr != NULL && !regex_search(full_file_name, re)))
284  {
285  continue;
286  }
287  else if ((file_data.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) != 0)
288  {
289  if (recursive)
290  {
291  GetAllFilesInDirectory(full_file_name, true, files);
292  }
293  continue;
294  }
295 
296  files.push_back(full_file_name);
297  } while (FindNextFileA(dir, &file_data));
298  FindClose(dir);
299 
300 #else
301 
302  class dirent* ent;
303  class stat st;
304  DIR* dir = opendir(directory.c_str());
305 
306  if (dir == NULL)
307  {
308  return false;
309  }
310 
311  while ((ent = readdir(dir)) != NULL)
312  {
313  const string file_name = ent->d_name;
314  const string full_file_name = directory + "/" + file_name;
315 
316  // if file is current path or does not exist (-1) then continue
317  if (file_name[0] == '.' || stat(full_file_name.c_str(), &st) == -1 || (rePtr != NULL && !regex_search(full_file_name, re)))
318  {
319  continue;
320  }
321  else if ((st.st_mode & S_IFDIR) != 0)
322  {
323  if (recursive)
324  {
325  GetAllFilesInDirectory(full_file_name, true, files);
326  }
327  continue;
328  }
329 
330  files.push_back(full_file_name);
331  }
332  closedir(dir);
333 
334 #endif
335 
336  return (files.size() != startSize);
337 }

424 uint64_t cISLogger::GetDirectorySpaceUsed(const string& directory, string regexPattern, vector<file_info_t>& files, bool sortByDate, bool recursive)
425 {
426  vector<string> fileNames;
427  GetAllFilesInDirectory(directory, recursive, regexPattern, fileNames);
428  uint64_t spaceUsed = 0;
429  for (unsigned int i = 0; i < fileNames.size(); i++)
430  {
431  file_info_t info;
432  info.name = fileNames[i];
433  struct stat st;
434  stat(info.name.c_str(), &st);
435  info.size = st.st_size;
436  info.lastModificationDate = st.st_mtime;
437  files.push_back(info);
438  spaceUsed += info.size;
439  }
440 
441  if (sortByDate)
442  {
443  struct
444  {
445  bool operator()(const file_info_t& a, const file_info_t& b)
446  {
447  int compare = (a.lastModificationDate == b.lastModificationDate ? 0 : (a.lastModificationDate < b.lastModificationDate ? -1 : 1));
448  if (compare == 0)
449  {
450  compare = a.name.compare(b.name);
451  }
452  return compare < 0;
453  }
454  } customSortDate;
455  sort(files.begin(), files.end(), customSortDate);
456  }
457  else
458  {
459  struct
460  {
461  bool operator()(const file_info_t& a, const file_info_t& b)
462  {
463  return a.name < b.name;
464  }
465  } customSortName;
466  sort(files.begin(), files.end(), customSortName);
467  }
468 
469  return spaceUsed;
470 }

521 bool cISLogger::LoadFromDirectory(const string& directory, eLogType logType, vector<string> serials)
522 {
523  // Delete and clear prior devices
524  Cleanup();
525  m_logType = logType;
526  string fileExtensionRegex;
527  set<string> serialNumbers;
528 
529  switch (logType)
530  {
531  default:
532  case cISLogger::LOGTYPE_DAT: fileExtensionRegex = "\\.dat$"; break;
533  case cISLogger::LOGTYPE_SDAT: fileExtensionRegex = "\\.sdat$"; break;
534  case cISLogger::LOGTYPE_CSV: fileExtensionRegex = "\\.csv$"; break;
535  case cISLogger::LOGTYPE_JSON: fileExtensionRegex = "\\.json$"; break;
536  case cISLogger::LOGTYPE_KML: return false; // fileExtensionRegex = "\\.kml$"; break; // kml read not supported
537  }
538 
539  // get all files, sorted by name
540  vector<file_info_t> files;
541  GetDirectorySpaceUsed(directory, fileExtensionRegex, files, false, false);
542  if (files.size() == 0)
543  {
544  return false;
545  }
546 
547  for (size_t i = 0; i < files.size(); i++)
548  {
549  string name = GetFileName(files[i].name);
550 
551  // check for log file prefix
552  size_t endOfLogPrefixIndex = name.find(IS_LOG_FILE_PREFIX);
553  if (endOfLogPrefixIndex != string::npos)
554  {
555  endOfLogPrefixIndex += IS_LOG_FILE_PREFIX_LENGTH;
556  size_t serialNumberEndIndex = name.find('_', endOfLogPrefixIndex);
557  if (serialNumberEndIndex != string::npos)
558  {
559  string serialNumber = name.substr(endOfLogPrefixIndex, serialNumberEndIndex - endOfLogPrefixIndex);
560 
561  // if we don't have a timestamp yet, see if we can parse it from the filename, i.e. IS_LOG_FILE_PREFIX 30013_20170103_151023_001
562  if (m_timeStamp.length() == 0)
563  {
564  // find _ at end of timestamp
565  size_t timestampIndex = name.find_last_of('_');
566  if (timestampIndex == string::npos)
567  {
568  timestampIndex = name.find_last_of('.');
569  }
570  if (timestampIndex != string::npos && timestampIndex - ++serialNumberEndIndex == IS_LOG_TIMESTAMP_LENGTH)
571  {
572  m_timeStamp = name.substr(timestampIndex - IS_LOG_TIMESTAMP_LENGTH, IS_LOG_TIMESTAMP_LENGTH);
573  }
574  }
575 
576  // check for unique serial numbers
577  if (serialNumbers.find(serialNumber) == serialNumbers.end())
578  {
579  if ((find(serials.begin(), serials.end(), "SN" + serialNumber) != serials.end() ||
580  find(serials.begin(), serials.end(), "ALL") != serials.end())) // and that it is a serial number we want to use
581  {
582  serialNumbers.insert(serialNumber);
583 
584  // Add devices
585  switch (logType)
586  {
587  default:
588  case cISLogger::LOGTYPE_DAT: m_devices.push_back(new cDeviceLogSerial()); break;
589  case cISLogger::LOGTYPE_SDAT: m_devices.push_back(new cDeviceLogSorted()); break;
590  case cISLogger::LOGTYPE_CSV: m_devices.push_back(new cDeviceLogCSV()); break;
591  case cISLogger::LOGTYPE_JSON: m_devices.push_back(new cDeviceLogJSON()); break;
592  }
593  m_devices.back()->SetupReadInfo(directory, serialNumber, m_timeStamp);
594  }
595  }
596  }
597  }
598  }
599 
600  for (uint32_t i = 0; i < m_devices.size(); i++)
601  {
602  m_devices[i]->InitDeviceForReading();
603  }
604  return (m_devices.size() != 0);
605 }

608 bool cISLogger::LogData(unsigned int device, p_data_hdr_t* dataHdr, const uint8_t* dataBuf)
609 {
610  m_lastCommTime = time(NULLPTR);
611 
612  if (!m_enabled)
613  {
614  if (m_errorFile != NULL)
615  {
616  fprintf(m_errorFile, "Logger is not enabled\r\n");
617  }
618  return false;
619  }
620  else if (device >= m_devices.size() || dataHdr == NULL || dataBuf == NULL)
621  {
622  if (m_errorFile != NULL)
623  {
624  fprintf(m_errorFile, "Invalid device handle or NULL data\r\n");
625  }
626  return false;
627  }
628  else if (LogHeaderIsCorrupt(dataHdr))
629  {
630  if (m_errorFile != NULL)
631  {
632  fprintf(m_errorFile, "Corrupt log header, id: %lu, offset: %lu, size: %lu\r\n", (unsigned long)dataHdr->id, (unsigned long)dataHdr->offset, (unsigned long)dataHdr->size);
633  }
634  m_logStats->LogError(dataHdr);
635  }
636  else if (!m_devices[device]->SaveData(dataHdr, dataBuf))
637  {
638  if (m_errorFile != NULL)
639  {
640  fprintf(m_errorFile, "Underlying log implementation failed to save\r\n");
641  }
642  m_logStats->LogError(dataHdr);
643  }
644  else
645  {
646  double timestamp = cISDataMappings::GetTimestamp(dataHdr, dataBuf);
647  m_logStats->LogDataAndTimestamp(dataHdr->id, timestamp);
648 
649  if (dataHdr->id == DID_DIAGNOSTIC_MESSAGE)
650  {
651  // write to diagnostic text file
652  std::ofstream outfile;
653  std::ostringstream outFilePath;
654  outFilePath << m_directory << "/diagnostic_" << m_devices[device]->GetDeviceInfo()->serialNumber << ".txt";
655  outfile.open(outFilePath.str(), std::ios_base::app);
656  std::string msg = (((diag_msg_t*)dataBuf)->message);
657  outfile.write(msg.c_str(), msg.length());
658  if (msg.length() > 0 && msg[msg.length() - 1] != '\n')
659  {
660  outfile << std::endl;
661  }
662  }
663  }
664  return true;
665 }

815 {

818  {

820  }
821  EnableLogging(true);
822  p_data_t* data = NULL;

824  {

827  SetDeviceInfo(devInfo, dev);

831 
832  // Copy data
833  while ((data = log.ReadData(dev)))
834  {
835  // CSV special cases
836  if (logType == eLogType::LOGTYPE_CSV)
837  {
838  if (data->hdr.id == DID_INS_2)
839  { // Convert INS2 to INS1 when creating .csv logs
840  ins_1_t ins1;
841  ins_2_t ins2;

862 bool cISLogger::ReadAllLogDataIntoMemory(const string& directory, map<uint32_t, vector<vector<uint8_t>>>& data)
863 {
864  cISLogger logger;
865  if (!logger.LoadFromDirectory(directory))
866  {
867  return false;
868  }
869  unsigned int deviceId = 0;
870  unsigned int lastDeviceId = 0xFFFFFEFE;
871  p_data_t* p;
872  vector<vector<uint8_t>>* currentDeviceData = NULL;
873  const dev_info_t* info;
874  uint8_t* ptr, *ptrEnd;
875  data.clear();
876 
877  // read every piece of data out of every device log
878  while ((p = logger.ReadNextData(deviceId)) != NULL)
879  {
880  // if this is a new device, we need to add the device to the map using the serial number
881  if (deviceId != lastDeviceId)
882  {
883  lastDeviceId = deviceId;
884  info = logger.GetDeviceInfo(deviceId);
885  data[info->serialNumber] = vector<vector<uint8_t>>();
886  currentDeviceData = &data[info->serialNumber];
887  }
888 
889  assert(currentDeviceData != NULL);
890 
891  // add slots until we have slots at least equal to the data id
892  while (currentDeviceData->size() < p->hdr.id)
893  {
894  currentDeviceData->push_back(vector<uint8_t>());
895  }
896 
897  // append each byte of the packet to the slot
898  ptrEnd = p->buf + p->hdr.size;
899  vector<uint8_t>& stream = (*currentDeviceData)[p->hdr.id];
900  for (ptr = p->buf; ptr != ptrEnd; ptr++)
901  {
902  stream.push_back(*ptr);
903  }
904  }
905  return true;
906 }

921 void cLogStatDataId::LogTimestamp(double timestamp)
922 {
923  // check for corrupt data
924  if (_ISNAN(timestamp) || timestamp < 0.0 || timestamp > 999999999999.0)
925  {
926  return;
927  }
928  else if (lastTimestamp > 0.0)
929  {
930  double delta = fabs(timestamp - lastTimestamp);
931  maxTimestampDelta = _MAX(delta, maxTimestampDelta);
932  totalTimeDelta += delta;
933  averageTimeDelta = (totalTimeDelta / (double)++timestampDeltaCount);
934  if (lastTimestampDelta != 0.0 && (fabs(delta - lastTimestampDelta) > (lastTimestampDelta * 0.5)))
935  {
936  timestampDropCount++;
937  }
938  lastTimestampDelta = delta;
939  }
940  lastTimestamp = timestamp;
941 }
