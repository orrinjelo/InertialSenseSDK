26 {

28  if (rb)

118 /*
119 * Given a ring buffer rb and a pointer to a location within its
120 * contiguous buffer, return the a pointer to the next logical
121 * location in the ring buffer.
122 */
123 static uint8_t* ringbuf_nextp(ringbuf_t* rb, const uint8_t *p)
124 {
125  /*
126  * The assert guarantees the expression (++p - rb->buf) is
127  * non-negative; therefore, the modulus operation is safe and
128  * portable.
129  */
130  assert((p >= rb->buf) && (p < ringbuf_end(rb)));
131  return rb->buf + ((++p - rb->buf) % ringbuf_buffer_size(rb));
132 }

134 size_t ringbuf_findchr(const ringbuf_t* rb, int c, size_t offset)
135 {
136  const uint8_t *bufend = ringbuf_end(rb);
137  size_t bytes_used = ringbuf_bytes_used(rb);
138  if (offset >= bytes_used)
139  return bytes_used;
140 
141  const uint8_t *start = rb->buf +
142  (((rb->tail - rb->buf) + offset) % ringbuf_buffer_size(rb));
143  assert(bufend > start);
144  size_t n = _MIN((size_t)(bufend - start), bytes_used - offset);
145  const uint8_t *found = memchr(start, c, n);
146  if (found)
147  return offset + (found - start);
148  else
149  return ringbuf_findchr(rb, c, offset + n);
150 }

152 size_t ringbuf_memset(ringbuf_t* dst, int c, size_t len)
153 {
154  const uint8_t *bufend = ringbuf_end(dst);
155  size_t nwritten = 0;
156  size_t count = _MIN(len, ringbuf_buffer_size(dst));
157  int overflow = count > ringbuf_bytes_free(dst);
158 
159  while (nwritten != count) {
160 
161  /* don't copy beyond the end of the buffer */
162  assert(bufend > dst->head);
163  size_t n = _MIN((size_t)(bufend - dst->head), count - nwritten);
164  memset(dst->head, c, n);
165  dst->head += n;
166  nwritten += n;
167 
168  // wrap?
169  if (dst->head == bufend)
170  dst->head = dst->buf;
171  }
172 
173  if (overflow) {
174  dst->tail = ringbuf_nextp(dst, dst->head);
175  assert(ringbuf_is_full(dst));
176  }
177 
178  return nwritten;
179 }

298 void* ringbuf_copy(ringbuf_t* dst, ringbuf_t* src, size_t count)
299 {
300  size_t src_bytes_used = ringbuf_bytes_used(src);
301  if (count > src_bytes_used)
302  return 0;
303  int overflow = count > ringbuf_bytes_free(dst);
304 
305  const uint8_t *src_bufend = ringbuf_end(src);
306  const uint8_t *dst_bufend = ringbuf_end(dst);
307  size_t ncopied = 0;
308  while (ncopied != count) {
309  assert(src_bufend > src->tail);
310  size_t nsrc = _MIN((size_t)(src_bufend - src->tail), count - ncopied);
311  assert(dst_bufend > dst->head);
312  size_t n = _MIN((size_t)(dst_bufend - dst->head), nsrc);
313  memcpy(dst->head, src->tail, n);
314  src->tail += n;
315  dst->head += n;
316  ncopied += n;
317 
318  // wrap?
319  if (src->tail == src_bufend)
320  src->tail = src->buf;
321  if (dst->head == dst_bufend)
322  dst->head = dst->buf;
323  }
324 
325  assert(count + ringbuf_bytes_used(src) == src_bytes_used);
326 
327  if (overflow) {
328  dst->tail = ringbuf_nextp(dst, dst->head);
329  assert(ringbuf_is_full(dst));
330  }
331 
332  return dst->head;
333 }
