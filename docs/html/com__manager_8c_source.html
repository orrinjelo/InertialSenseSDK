426 void comManagerStepInstance(CMHANDLE cmInstance_)
427 {
428  com_manager_t* cmInstance = (com_manager_t*)cmInstance_;
429  uint8_t c, additionalDataAvailable, canReadAgain;
430  int32_t n, pHandle, freeByteCount;
431  ring_buffer_t* ringBuffer;
432 
433  if (cmInstance->readCallback)
434  {
435  for (pHandle = 0; pHandle < cmInstance->numHandes; pHandle++)
436  {
437  ringBuffer = cmInstance->ringBuffers + pHandle;
438  canReadAgain = 1;
439 
440  readAgain:
441 
442  freeByteCount = (RING_BUFFER_SIZE - ringBuffer->endIndex);
443 
444  // if we are out of free space, we need to either move bytes over or start over
445  if (freeByteCount == 0)
446  {
447  if (ringBuffer->startIndex < RING_BUFFER_FLUSH_THRESHOLD)
448  {
449  // we will be hung unless we flush the ring buffer, we have to drop bytes in this case and the caller
450  // will need to resend the data
451  ringBuffer->startIndex = 0;
452  ringBuffer->endIndex = 0;
453  ringBuffer->scanIndex = 0;
454  }
455  else
456  {
457  // shift over the remaining data in the hopes that we will get a valid packet by appending the next read call
458  ringBuffer->endIndex -= ringBuffer->startIndex;
459  ringBuffer->scanIndex -= ringBuffer->startIndex;
460  memmove(ringBuffer->buf, ringBuffer->buf + ringBuffer->startIndex, ringBuffer->endIndex);
461  ringBuffer->startIndex = 0;
462  }
463 
464  // re-calculate free byte count
465  freeByteCount = (RING_BUFFER_SIZE - ringBuffer->endIndex);
466  }
467 
468  if ((n = cmInstance->readCallback(cmInstance, pHandle, ringBuffer->buf + ringBuffer->endIndex, freeByteCount)) > 0)
469  {
470  additionalDataAvailable = 0;
471  cmInstance->status[pHandle].readCounter += n;
472  ringBuffer->endIndex += n;
473 
474  while (ringBuffer->scanIndex < ringBuffer->endIndex)
475  {
476  // check for pass through bytes
477  if (cmInstance->status[pHandle].passThroughBytes > 0)
478  {
479  processPassThroughBytes(cmInstance, pHandle, ringBuffer);
480  continue;
481  }
482 
483  c = ringBuffer->buf[ringBuffer->scanIndex++];
484 
485  switch (c)
486  {
487  case PSC_ASCII_START_BYTE:
488  {
489  cmInstance->status[pHandle].startByte = PSC_ASCII_START_BYTE;
490  ringBuffer->startIndex = ringBuffer->scanIndex - 1;
491  } break;
492 
493  case PSC_START_BYTE:
494  {
495  cmInstance->status[pHandle].startByte = PSC_START_BYTE;
496  ringBuffer->startIndex = ringBuffer->scanIndex - 1;
497  } break;
498 
499  case PSC_ASCII_END_BYTE:
500  {
501  if (cmInstance->status[pHandle].startByte == PSC_ASCII_START_BYTE)
502  {
503  processAsciiPacket(cmInstance, pHandle, ringBuffer->buf + ringBuffer->startIndex, ringBuffer->scanIndex - ringBuffer->startIndex);
504  }
505  ringBuffer->startIndex = ringBuffer->scanIndex;
506  cmInstance->status[pHandle].startByte = 0;
507  } break;
508 
509  case PSC_END_BYTE:
510  {
511  if (cmInstance->status[pHandle].startByte == PSC_START_BYTE)
512  {
513  processBinaryPacket(cmInstance, pHandle, ringBuffer->buf, ringBuffer->buf + ringBuffer->startIndex, ringBuffer->scanIndex - ringBuffer->startIndex, &additionalDataAvailable);
514  }
515  ringBuffer->startIndex = ringBuffer->scanIndex;
516  cmInstance->status[pHandle].startByte = 0;
517  } break;
518 
519  case UBLOX_START_BYTE1:
520  {
521  if (beginUbloxPacket(cmInstance, pHandle, ringBuffer))
522  {
523  return;
524  }
525  } break;
526 
527  case RTCM3_START_BYTE:
528  {
529  if (beginRtcm3Packet(cmInstance, pHandle, ringBuffer))
530  {
531  return;
532  }
533  } break;
534  }
535  }
536 
537  if (additionalDataAvailable && canReadAgain)
538  {
539  canReadAgain = 0;
540  goto readAgain;
541  }
542  }
543 
544  if ((cmInstance->status[pHandle].flags & CM_PKT_FLAGS_RX_VALID_DATA) && cmInstance->status[pHandle].readCounter > 128)
545  {
546  // communication problem, clear communication received bit
547  cmInstance->status[pHandle].flags &= (~CM_PKT_FLAGS_RX_VALID_DATA);
548  }
549  }
550  }
551 
552  stepComManagerSendMessagesInstance(cmInstance);
553 }

560 void stepComManagerSendMessagesInstance(CMHANDLE cmInstance_)
561 {
562  com_manager_t* cmInstance = cmInstance_;
563 
564  // Send data (if necessary)
565  for (broadcast_msg_t* bcPtr = cmInstance->broadcastMessages, *ptrEnd = (cmInstance->broadcastMessages + MAX_NUM_BCAST_MSGS); bcPtr < ptrEnd; bcPtr++)
566  {
567  // If send buffer does not have space, exit out
568  if (cmInstance->txFreeCallback && (bcPtr->pkt.txData.size > (uint32_t)cmInstance->txFreeCallback(cmInstance, bcPtr->pHandle)))
569  {
570  break;
571  }
572  // Send once and remove from message queue
573  else if (bcPtr->period == MSG_PERIOD_SEND_ONCE)
574  {
575  sendDataPacket(cmInstance, bcPtr->pHandle, &(bcPtr->pkt));
576  disableBroadcastMsg(cmInstance, bcPtr);
577  }
578  // Broadcast messages
579  else if (bcPtr->period > 0)
580  {
581  // Check if counter has expired
582  if (++bcPtr->counter >= bcPtr->period)
583  {
584  bcPtr->counter = 0; // reset counter
585 
586  // Prep data if callback exists
587  if (cmInstance->regData[bcPtr->dataHdr.id].preTxFnc)
588  {
589  cmInstance->regData[bcPtr->dataHdr.id].preTxFnc(cmInstance, bcPtr->pHandle);
590  }
591  sendDataPacket(cmInstance, bcPtr->pHandle, &(bcPtr->pkt));
592  }
593  }
594  }
595 
596  // Resend data (if necessary)
597  stepPacketRetry(cmInstance);
598 }

882 {

914 void parseAsciiPacket(com_manager_t* cmInstance, int pHandle, unsigned char* buf, int count)
915 {
916  unsigned char c;
917  unsigned char* messageId;
918  unsigned char* messageData = 0;
919  unsigned char* end = buf + count;
920  unsigned char* ptr = buf;
921  int fieldIndex = 0;
922  int length;
923  asciiMessageMap_t* foundMap = 0;
924  cmInstance->status[pHandle].flags |= CM_PKT_FLAGS_RX_VALID_DATA; // communication received
925  cmInstance->status[pHandle].readCounter = 0;
926 
927  // Packet read success
928  cmInstance->status[pHandle].rxPktCount++;
929 
930  // message id is right after the $ char
931  messageId = ++ptr;
932 
933  // in order to parse the fields from the ascii message, we need to not have a global handler and
934  // have at least one ascii message defined and
935  // have a map for this message id
936  if
937  (
938  (cmInstance->asciiMessageHandler == 0 || cmInstance->asciiMessageHandler(cmInstance, pHandle, messageId, buf, count) == 0) &&
939  (cmInstance->asciiMessages != 0 && cmInstance->asciiMessagesCount > 0) &&
940  ((foundMap = bsearch(messageId, cmInstance->asciiMessages, cmInstance->asciiMessagesCount, sizeof(cmInstance->asciiMessages[0]), findAsciiMessage)) != 0)
941  )
942  {
943  while (ptr < end)
944  {
945  c = *ptr;
946  if (c == ',' || c == '*')
947  {
948  // check if we have the first piece of data yet and have not run out of fields
949  if (fieldIndex < foundMap->fieldCount && messageData != 0)
950  {
951  // if we have a non-empty message, parse it out
952  if ((length = (int)(ptr - messageData)) > 0)
953  {
954  *ptr = '\0';
955  uint16_t fieldAndOffset = foundMap->fieldsAndOffsets[fieldIndex];
956  asciiDataType type = (fieldAndOffset & 0x00FF);
957  uint8_t offset = (fieldAndOffset & 0xFF00) >> 8;
958  switch (type)
959  {
960  case asciiTypeInt:
961  {
962  const int dataValue = strtol((char*)messageData, 0, 10);
963  memcpy(foundMap->ptr + offset, &dataValue, sizeof(dataValue));
964  } break;
965  case asciiTypeUInt:
966  {
967  const unsigned int dataValue = strtoul((char*)messageData, 0, 10);
968  memcpy(foundMap->ptr + offset, &dataValue, sizeof(dataValue));
969  } break;
970  case asciiTypeFloat:
971  {
972  const float dataValue = PARSE_FLOAT((char*)messageData);
973  memcpy(foundMap->ptr + offset, &dataValue, sizeof(dataValue));
974  } break;
975  case asciiTypeDouble:
976  {
977  const double dataValue = PARSE_DOUBLE((char*)messageData);
978  memcpy(foundMap->ptr + offset, &dataValue, sizeof(dataValue));
979  } break;
980  }
981  *ptr = c;
982  }
983  fieldIndex++;
984  }
985 
986  // message data is at next char
987  messageData = ++ptr;
988  }
989  else
990  {
991  ptr++;
992  }
993  }
994  }
995 
996  // send post ascii parse message if set
997  if (cmInstance->asciiMessageHandlerPost != 0)
998  {
999  cmInstance->asciiMessageHandlerPost(cmInstance, pHandle, messageId, buf, count);
1000  }
1001 }
